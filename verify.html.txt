<script>

async function verifyBatch(){

  const params =
    new URLSearchParams(window.location.search);

  const batchCode = params.get("batch");

  const { data, error } =
    await supabase
      .from("batches")
      .select("*")
      .eq("batch_code", batchCode)
      .single();

  const result =
    document.getElementById("result");

  if(error || !data){

    result.innerHTML = `
      <p class="invalid">
        ❌ Invalid Honey Batch
      </p>
    `;

    return;
  }

  result.innerHTML = `
    <p class="verified">
      ✅ Verified Honey Batch
    </p>

    <br>

    <p>
      <strong>Batch Code:</strong>
      ${data.batch_code}
    </p>

    <p>
      <strong>Honey Type:</strong>
      ${data.honey_type}
    </p>

    <p>
      <strong>Harvest Date:</strong>
      ${data.harvest_date}
    </p>

    <p>
      <strong>Status:</strong>
      ${data.verified ? "Verified" : "Pending"}
    </p>
  `;
}

verifyBatch();

</script>
